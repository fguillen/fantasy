def initialize(kind:, position:, width:, height:, stroke: 1, fill: true, color: Color.palette.black, stroke_color: nil)
  @kind = kind
  @position = position
  @width = width
  @height = height
  @stroke = stroke
  @color = color
  @fill = fill
  @stroke_color = stroke_color
  @layer = 1

  Global.shapes << self
end

  end
