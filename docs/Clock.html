def repeat(seconds: 1, times: Float::INFINITY)
  times_executed = 0
  @thread =
    Thread.new do
      while times_executed < times
        @block.call
        times_executed += 1

        seconds_to_sleep = seconds.is_a?(Range) ? rand(seconds) : seconds
        sleep(seconds_to_sleep)
      end
    end
end
