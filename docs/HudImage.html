def initialize(position:, image_name:)
  @image = Image.new(image_name)
  @name = image_name
  @position = position
  @scale = 1
  @visible = true
  @draggable_on_debug = true
  @dragging = false
  @layer = 100

  Global.hud_images.push(self)
end

def draw
  if visible
    @image.draw(x: @position.x, y: @position.y, scale: @scale)
  end

  draw_debug if Global.debug
end
