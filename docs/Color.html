def hex
  [@r, @g, @b].map do |e|
    e.to_s(16).rjust(2, "0")
  end.join
end

def to_s
  result = "r:#{@r}, g:#{@g}, b:#{@b}, a:#{@a}, hex:#{hex}"
  result += " (#{name})" unless name.nil?

  result
end
